Perfect — now that your diff detection and saving workflow works, you can follow this step-by-step workflow to apply .diff patch files in Git.

2️⃣ Add initial files and push

# Add new files
git add .

# Commit changes
git commit -m "Initial files"

# Push to GitHub
git push origin main

// ==UserScript==
// @name         ChatGPT Diff Saver (Fixed & Robust)
// @namespace    https://songdrop.dev/chatgpt-diff-saver
// @version      0.2.0
// @description  Detect GitHub-style diffs in ChatGPT outputs (including ```diff blocks) and save as .diff with floating buttons.
// @match        https://chat.openai.com/*
// @match        https://chatgpt.com/*
// @grant        none
// ==/UserScript==

(function() {
    'use strict';

    // Regex to detect GitHub-style diffs anywhere in text
    const DIFF_HEADER = /diff --git a\/.+ b\/.+/;

    function saveDiff(diffText) {
        let filename = prompt('Filename (with .diff extension):', `gpt-${Date.now()}.diff`);
        if (!filename) return;
        if (!filename.endsWith('.diff')) filename += '.diff';

        const blob = new Blob([diffText], { type: 'text/x-diff' });
        const url = URL.createObjectURL(blob);

        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        a.style.display = 'none';
        document.body.appendChild(a);
        a.dispatchEvent(new MouseEvent('click'));
        document.body.removeChild(a);
        URL.revokeObjectURL(url);
    }

    // Styles for floating button like native Copy Code
    const style = document.createElement('style');
    style.textContent = `
        .diff-save-btn-floating {
            position: absolute;
            top: 6px;
            left: 60px;
            font-size: 12px;
            padding: 4px 6px;
            border-radius: 6px;
            border: none;
            cursor: pointer;
            background: rgba(0,0,0,0.2);
            color: #fff;
            display: flex;
            align-items: center;
            gap: 4px;
            transition: background 0.15s, transform 0.15s;
            z-index: 10;
        }
        .diff-save-btn-floating:hover {
            background: rgba(0,0,0,0.35);
            transform: translateY(-1px);
        }
        .diff-pre-container {
            position: relative;
        }
        .diff-save-row {
            display: flex;
            align-items: center;
            gap: 10px;
            margin-top: 6px;
        }
    `;
    document.head.appendChild(style);

    function scan() {
        document.querySelectorAll('pre').forEach(pre => {
            if (pre.dataset.diffChecked) return;

            // Get text and clean it up
            let text = (pre.textContent || '').trim();

            // Remove Markdown-style ```diff wrapping if present
            text = text.replace(/^```diff\s*|```$/g, '').trim();

            if (!DIFF_HEADER.test(text)) return;

            pre.dataset.diffChecked = 'true';

            // Highlight detected diff
            pre.style.outline = '3px solid red';

            // Wrap pre in relative container for floating button
            const container = document.createElement('div');
            container.className = 'diff-pre-container';
            pre.parentNode.insertBefore(container, pre);
            container.appendChild(pre);

            // Floating button (top-left)
            const btnFloating = document.createElement('button');
            btnFloating.className = 'diff-save-btn-floating';
            btnFloating.innerHTML = '📄 Save .diff';
            btnFloating.onclick = () => saveDiff(text);
            container.appendChild(btnFloating);

            // Bottom row button with label
            const row = document.createElement('div');
            row.className = 'diff-save-row';

            const label = document.createElement('strong');
            label.textContent = 'DIFF DETECTED';
            label.style.fontSize = '12px';

            const btnBottom = document.createElement('button');
            btnBottom.textContent = 'Save .diff';
            btnBottom.style.fontSize = '12px';
            btnBottom.style.padding = '4px 8px';
            btnBottom.style.borderRadius = '6px';
            btnBottom.style.cursor = 'pointer';
            btnBottom.onclick = () => saveDiff(text);

            row.appendChild(label);
            row.appendChild(btnBottom);

            pre.after(row);
        });
    }

    // Polling for dynamically loaded code blocks
    setInterval(scan, 1000);
})();
