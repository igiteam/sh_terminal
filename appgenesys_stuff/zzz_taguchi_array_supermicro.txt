  {"id":3,"electron":"27.0","npm":"yarn","fix":"skip_boot"},

  {"id":8,"electron":"29.0","npm":"ci","fix":"skip_boot"},
  {"id":9,"electron":"29.0","npm":"yarn","fix":"none"}
]
EOF
